\chapter*{Anhang}\label{ch:anhang}

Der Anhang dient zur Aufnahme ergänzender Informationen, die den Lesefluss im Haupttext stören würden, aber zur Nachvollziehbarkeit beitragen.

\chapter*{Nützliche Befehle}

\section{Zitation mit biblatex}
Zitiere im Text mit \verb|\autocite{schluessel}| oder \verb|\textcite{schluessel}|. 
Die Bibliographie wird am Dokumentende mit \verb|\printbibliography| ausgegeben.
Beispiel: Die Methode folgt \cite{knuth1984texbook}.

\section{Mathematische Formeln (amsmath, siunitx)}
Skalare: $a\in\mathbb{R}$, Vektoren: $\mathbf{x}\in\mathbb{R}^{n}$, Matrizen: $\mathbf{A}\in\mathbb{R}^{m\times n}$. Einheiten mit \texttt{siunitx}:
Leistung $P=\SI{10}{\watt}$, Frequenz $f=\SI{1}{\kilo\hertz}$.
\begin{align}
\mathbf{y} &= \mathbf{A}\mathbf{x} + \boldsymbol{\varepsilon},\\
\mathcal{L}\{x(t)\} &= X(s) = \int_{0}^{\infty} x(t)\,e^{-st}\,\mathrm{d}t.
\end{align}
