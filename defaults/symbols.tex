\subsection*{Symbole}
\begin{itemize}[label=$\bullet$]
    \item \textbf{$a, b, c$}: Skalare bzw. Konstanten
    \item \textbf{$\mathbf{a}, \mathbf{x}$}: Vektoren (Spaltenvektoren)
    \item \textbf{$\mathbf{A}, \mathbf{B}$}: Matrizen
    \item \textbf{$\mathbb{R}$}: Menge der reellen Zahlen
    \item \textbf{$\mathbb{N}$}: Menge der natürlichen Zahlen
    \item \textbf{$\mathbb{Z}$}: Menge der ganzen Zahlen
    \item \textbf{$\mathbb{C}$}: Menge der komplexen Zahlen
    \item \textbf{$\det(\mathbf{A})$}: Determinante der Matrix $\mathbf{A}$
    \item \textbf{$\mathbf{A}^{-1}$}: Inverse einer Matrix $\mathbf{A}$
    \item \textbf{$\mathbf{A}^\mathsf{T}$}: Transponierte Matrix
    \item \textbf{$\mathbf{A}^\mathsf{H}$}: Hermitesch Transponierte (konjugiert-transponiert)
    \item \textbf{$\|\mathbf{x}\|$}: Norm eines Vektors $\mathbf{x}$
    \item \textbf{$\langle \mathbf{x}, \mathbf{y} \rangle$}: Skalarprodukt von $\mathbf{x}$ und $\mathbf{y}$
    \item \textbf{$\nabla f$}: Gradient der Funktion $f$
    \item \textbf{$\partial f / \partial x$}: Partielle Ableitung von $f$ nach $x$
    \item \textbf{$\mathrm{d}x$}: Differential eines Skalars $x$
    \item \textbf{$\int f(x)\,\mathrm{d}x$}: Integral der Funktion $f$
    \item \textbf{$\sum_{i=1}^{n} a_i$}: Summe der Terme $a_i$
    \item \textbf{$\prod_{i=1}^{n} a_i$}: Produkt der Terme $a_i$
    \item \textbf{$\delta_{ij}$}: Kronecker-Delta
    \item \textbf{$\varepsilon$}: Kleine positive Zahl (Fehler, Toleranz)
    \item \textbf{$\mu$}: Erwartungswert einer Zufallsvariablen
    \item \textbf{$\sigma$}: Standardabweichung
    \item \textbf{$\Sigma$}: Kovarianzmatrix
    \item \textbf{$\lambda$}: Eigenwert bzw. Regularisierungsparameter
    \item \textbf{$\phi, \varphi$}: Winkel oder allgemeine Abbildung
    \item \textbf{$\theta$}: Parameter (z.\,B. im Modell oder in Statistik)
    \item \textbf{$\omega$}: Kreisfrequenz
    \item \textbf{$\Omega$}: Definitionsmenge oder Frequenzbereich
    \item \textbf{$\Delta$}: Differenz oder Änderung ($\Delta x = x_{t+1}-x_t$)
    \item \textbf{$\alpha, \beta, \gamma$}: Koeffizienten, Konstanten oder Parameter
    \item \textbf{$\mathcal{L}\{x(t)\}$}: Laplace-Transformation von $x(t)$
    \item \textbf{$\mathcal{F}\{x(t)\}$}: Fourier-Transformation von $x(t)$
    \item \textbf{$E[\cdot]$}: Erwartungswertoperator
    \item \textbf{$\mathrm{Var}[\cdot]$}: Varianzoperator
    \item \textbf{$P(A)$}: Wahrscheinlichkeit des Ereignisses $A$
    \item \textbf{$p(x)$}: Wahrscheinlichkeitsdichtefunktion
    \item \textbf{$f(x)$}: Allgemeine Funktion
    \item \textbf{$t$}: Zeitvariable
    \item \textbf{$k, n, m$}: Lauf- oder Indexvariablen
\end{itemize}
